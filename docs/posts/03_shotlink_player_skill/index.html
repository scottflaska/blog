strokes_remaining <- shot_level %>% 
  group_by(player,
           hole,
           round) %>% 
  arrange(player,
          round,
          hole,
          shot) %>% 
  mutate(strokes_rolling = cumsum(num_of_strokes)) %>% 
  mutate(strokes_remaining = hole_score - strokes_rolling + 1) %>%
  ungroup()

strokes_remaining %>%

         hole %in% 1) %>%

         hole_score,
         shot,
         num_of_strokes,
